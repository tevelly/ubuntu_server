Dodavanje korisnika
# useradd –c "Testni user" –m –s /bin/bash –g grupa –G dodGrupa iperic
-c komentar, -m kreiranje home direktorija, -s putanja do korisnikovog shell-a
-g dodavanje u primarnu (default) grupu, -G dodatne grupe
Brisanje korisnika :
#sudo userdel –r username //briše korisnika i home dir
Brisanje korisničkog direktorija :
#sudo rm -r /home/username ili
#sudo deluser --remove-home user

Upravljanje korisnicima i grupama
U /etc/group se nalaze sve grupe
root:x:0:
korisnici:x:1001:ivan,josipa
Format /etc/group datoteke:
ime_grupe:password:GID:account1,accountN

root grupa nema listu korisnika

Dodavanje grupe
groupadd [ options ] group_name
# groupadd posluzitelj
# tail -1 /etc/group posluzitelj:x:1003:
# groupadd -g 2500 baza 
# tail -1 /etc/group baza:x:2500:
Brisanje grupe
groupdel group_name
# groupdel posluzitelj (obratiti pozornost na primarnu grupu)

Dodavanje postojećih korisnika u grupu
#sudo usermod -a -G group username
Brisanje postojećih korisnika iz grupe
#sudo deluser username groupname
Preimenovanje postojeće grupe
groupmod -n novoImeGrupe staroImeGrupe
- n MAJSTORI 	Promjena imena grupe

Izlistavanje postojećih grupa
more /etc/group
Izlistavanje postojećih korisnika s priradajućim grupama
groups ime_korisnika

Dozvole
vlasnik:grupa:ostali
Mijenjanje prava pristupa
chmod -R 777 file/folder

Mijenjanje vlasništva file/folder
chown -R root:sistemci file/folder
Mijenjanje vlasništva file/folder grupe
chgrp -R groupname file/folder
